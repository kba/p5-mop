=head1 NAME

Perl 5 MOP - Syntax & Semantics - Introduction

=head1 SYNOPSIS

  class Point {
      has $x = 0;
      has $y = 0;

      method clear {
          ($x, $y) = (0, 0);
      }
  }

  class Point3D ( extends => Point ) {
      has $z = 0;

      method clear {
          super;
          $z = 0;
      }
  }

  my $p = Point->new( x => 10, y => 10 );

  my $p3d = Point3D->new( x => 10, y => 10, z => 1 );
